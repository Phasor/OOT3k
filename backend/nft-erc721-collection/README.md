# Deploy
1. Get truffle dashboard up with `truffle dashboard`
2. yarn deploy --network truffle

# Verify on etherscan
yarn verify [contract address] --network truffle

# Open whitelist
yarn whitelist-open --network truffle

# Close whitelist
yarn whitelist-close --network truffle

# Open public sales
yarn public-sale-open --network truffle
